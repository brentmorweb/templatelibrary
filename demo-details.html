 * Modern Hero Section — JS
 * hero-section.js
 */

(function() {
  'use strict';

  const hero = document.querySelector('.hero-section');
  if (!hero) return;

  // Animate hero content on load
  const content = hero.querySelector('.hero-content');
  const image = hero.querySelector('.hero-image');
  const features = hero.querySelectorAll('.hero-feature-item');

  function animateIn(el, delay = 0) {
    el.style.cssText += `
      opacity: 0;
      transform: translateY(20px);
      transition: opacity 0.5s ease ${delay}ms,
                  transform 0.5s ease ${delay}ms;
    `;
    requestAnimationFrame(() => {
      requestAnimationFrame(() => {
        el.style.opacity = '1';
        el.style.transform = 'translateY(0)';
      });
    });
  }

  if (content) animateIn(content, 0);
  if (image)   animateIn(image,   100);

  features.forEach((el, i) => {
    animateIn(el, 200 + i * 80);
  });

  // Parallax effect on scroll
  const heroImg = hero.querySelector('.hero-image img');
  if (heroImg) {
    window.addEventListener('scroll', () => {
      const rect = hero.getBoundingClientRect();
      const offset = (rect.top / window.innerHeight) * 20;
      heroImg.style.transform = `translateY(${offset}px)`;
    }, { passive: true });
  }

})();
